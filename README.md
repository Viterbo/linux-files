# install dev tools

```sh
cd /tmp
wget https://raw.githubusercontent.com/Viterbo/linux-files/master/update_ubuntu.sh

sudo chmod +x update_ubuntu.sh
sudo chmod +x install_docker.sh
sudo chmod +x install_eos_docker.sh
./update_ubuntu.sh
```
